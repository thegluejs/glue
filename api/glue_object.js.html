'use strict';

const GlueException = require('./exception');

/**
 * @class
 * @namespace GlueException
 */
class GlueObject {

  /**
   * @constructor
   */
  constructor() {

  }

  /**
   * @static
   * @method
   * @this {GlueNamespace}
   * @param {String} name - The name that will be checked
   * @return {Boolean} If a name is valid or not
   * @example
   * if (!this.isValid()) {
   *   throw GlueException('The given name is not valid!');
   * }
   */
  isValidName (name) {
    return /^[A-Za-z-:]*$/.test(name);
  }

  /**
   * @static
   * @method
   * @this {GlueNamespace}
   * @param {String} name - The name that will be checked
   * @param {GlueInstance} instance - The GlueInstance to be check against
   * @return {Boolean} If a name is registered or not
   * @example
   *
   * if(this.isRegistered()) {
   *   throw new GlueException('The given name is already registered!')
   * }
   */
   isRegisteredNamespace (name, instance) {

    /*
     First check if there is a namespace registered
     */
    if (instance.namespaces.length > 0) {
      for (let namespace of instance.namespaces) {
        if (namespace.name === name) {
          throw new GlueException('There is a namespace a registered with this name: ' + name);
        }
      }
    }

    return false;
  };
}

 */
module.exports = GlueObject;
